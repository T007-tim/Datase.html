Exploratory Data Analysis on Iris Dataset import pandas as pd from sk learn.datasets import load_iris # Load the Iris Dataset iris = load_iris() iris_df = pd.DataFrame(data=iris.data, columns=iris.feature_names) iris_df['target'] = iris.target # Display the first few rows of the dataset print(iris_df.head()) # Display basic statistics of the dataset print(iris_df.describe()) # Group by species and compute mean for numerical feature_names print(iris_df.groupby('target').mean()) # Check for missing values print(iris_df.isnull().sum())
